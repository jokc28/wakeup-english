> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Replace 옥모닝's brand identity (color, typography, mascot, app icon) and redesign 4 key screens (Home, Alarm Edit, Paywall, Settings) for a Modern Playful aesthetic, before the first Play/App Store submission.

**Architecture:** Flutter app with riverpod state. Visual layer is centralized in `lib/core/constants/` (tokens) and `lib/app.dart` (theme). Mascot is shipped as 5 SVG files (one per expression) loaded via `flutter_svg`. Brand assets (app icons, feature graphic) are generated by Python scripts that are reproducible and committed alongside the source SVGs.

**Tech Stack:** Flutter 3.38, Dart 3.10, riverpod 2.6, flutter_svg, flutter_animate, Pretendard font, Python 3 + Pillow + rsvg-convert (build-time asset generation only).

**Spec:** `docs/superpowers/specs/2026-05-15-okmorning-visual-refresh-design.md`

- [ ] **Step 3.1: Write download script**

Create `scripts/download-pretendard.sh`:

```bash
#!/usr/bin/env bash
# Downloads the 5 Pretendard TTF weights we ship into assets/fonts/.
# Pretendard is OFL 1.1 — bundling and redistribution is permitted.

set -euo pipefail

VERSION="1.3.9"
BASE="https://github.com/orioncactus/pretendard/raw/v${VERSION}/packages/pretendard/dist/public/static"
OUT_DIR="assets/fonts"
mkdir -p "$OUT_DIR"

for weight in Regular Medium SemiBold Bold ExtraBold; do
  url="$BASE/Pretendard-$weight.otf"
  out="$OUT_DIR/Pretendard-$weight.ttf"
  # The repo ships .otf; rename to .ttf locally — Flutter accepts both formats
  # under either extension as long as pubspec declarations match the asset
  # path.
  echo "Fetching $weight..."
  curl -sSL "$url" -o "$out"
done

ls -la "$OUT_DIR"
```

```dart
import 'package:flutter/material.dart';
import 'package:flutter_svg/flutter_svg.dart';

enum SunnyExpression { smile, wink, sad, sleepy, excited }

extension SunnyExpressionAsset on SunnyExpression {
  String get assetPath => 'assets/mascot/sunny-$name.svg';
}

/// Renders the Sunny mascot SVG at a given size.  Use this everywhere the
/// brand mascot appears.  Wrap in your own animation widget if motion is
/// needed (no implicit animation here — keep this pure).
class Sunny extends StatelessWidget {
  const Sunny({
    super.key,
    required this.expression,
    this.size = 64,
    this.semanticLabel,
  });

  final SunnyExpression expression;
  final double size;
  final String? semanticLabel;

  @override
  Widget build(BuildContext context) {
    return SvgPicture.asset(
      expression.assetPath,
      key: Key('sunny-${expression.name}'),
      width: size,
      height: size,
      semanticsLabel: semanticLabel ?? 'Sunny ${expression.name}',
    );
  }
}
```

```python
#!/usr/bin/env python3
"""Generates the Sunny-centered app icon at 1024×1024, then renders all
iOS / Android / Play sizes from that master.

    python3 scripts/generate-app-icon.py

Flags:
    --out-master PATH     where to write the 1024 master (default:

    --out-play   PATH     where to write the 512 Play listing icon

from __future__ import annotations

import argparse
import subprocess
import tempfile
from pathlib import Path

REPO = Path(__file__).resolve().parents[1]

ICON_SVG_TEMPLATE = """<svg xmlns="http://www.w3.org/2000/svg" viewBox="0 0 1024 1024">

</svg>"""


def render(svg_path: Path, png_path: Path, target_px: int) -> None:
    subprocess.run(
        ["rsvg-convert", "-w", str(target_px), "-h", str(target_px),
         "-o", str(png_path), str(svg_path)],
        check=True,
    )


def main() -> None:
    parser = argparse.ArgumentParser()
    parser.add_argument("--out-master", default=str(
        REPO / "ios" / "Runner" / "Assets.xcassets" / "AppIcon.appiconset" / "Icon-App-1024x1024@1x.png"))
    parser.add_argument("--out-play", default=str(REPO / "dist" / "play-icon-512.png"))
    parser.add_argument("--skip-platforms", action="store_true")
    args = parser.parse_args()

    with tempfile.NamedTemporaryFile(suffix=".svg", delete=False) as tf:
        tf.write(ICON_SVG_TEMPLATE.encode("utf-8"))
        svg_path = Path(tf.name)

    master = Path(args.out_master)
    play = Path(args.out_play)
    master.parent.mkdir(parents=True, exist_ok=True)
    play.parent.mkdir(parents=True, exist_ok=True)

    render(svg_path, master, 1024)
    render(svg_path, play, 512)

            "Icon-App-60x60@2x.png": 120, "Icon-App-60x60@3x.png": 180,
            "Icon-App-76x76@1x.png": 76,  "Icon-App-76x76@2x.png": 152,
            "Icon-App-83.5x83.5@2x.png": 167,
        }
        for name, px in ios_sizes.items():
            render(svg_path, ios_dir / name, px)

        # Android legacy mipmaps
        android_sizes = {
            "mipmap-mdpi":    48,
            "mipmap-hdpi":    72,
            "mipmap-xhdpi":   96,
            "mipmap-xxhdpi":  144,
            "mipmap-xxxhdpi": 192,

        for folder, px in android_sizes.items():

            target.parent.mkdir(parents=True, exist_ok=True)
            render(svg_path, target, px)

    svg_path.unlink()
    print(f"Master: {master}")
    print(f"Play:   {play}")
    if not args.skip_platforms:
        print("iOS density set and Android mipmaps regenerated in place.")


if __name__ == "__main__":
    main()
```

```dart
import 'package:flutter/material.dart';

import '../../core/widgets/sunny.dart';

class SunnyFeedbackController extends ChangeNotifier {
  SunnyExpression? _active;
  SunnyExpression? get active => _active;

  void showCorrect() => _flash(SunnyExpression.wink);
  void showWrong() => _flash(SunnyExpression.sad);

  Future<void> _flash(SunnyExpression expression) async {
    _active = expression;
    notifyListeners();
    await Future.delayed(const Duration(milliseconds: 800));
    _active = null;
    notifyListeners();
  }
}

class SunnyFeedbackOverlay extends StatelessWidget {
  const SunnyFeedbackOverlay({
    super.key,
    required this.controller,
    required this.child,
  });

  final SunnyFeedbackController controller;
  final Widget child;

  @override
  Widget build(BuildContext context) {
    return Stack(
      children: [
        child,
        AnimatedBuilder(
          animation: controller,
          builder: (_, __) {
            final exp = controller.active;
            if (exp == null) return const SizedBox.shrink();
            return Center(
              child: AnimatedScale(
                scale: 1.0,
                duration: const Duration(milliseconds: 220),
                child: Sunny(expression: exp, size: 160),
              ),
            );
          },
        ),
      ],
    );
  }
}
```
